;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; 宝箱のテスト
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
Begin
	prologue
	fadein	0.0
End



;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; 宝箱を開けるイベント
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
Begin
	target		Hero
	target		TreasureBox
	condition	TreasureBox		isOpened	0

	message	TreasureBox		open	;; 宝箱を開ける

	show	TreasureKey				;; 鍵を表示する

	play	rpg_system08
	text	中から鍵が出てきた！

	set		TreasureBox		isOpened	1
End

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; 鍵を拾うイベント
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
Begin
	target	Hero
	target	TreasureKey

	play	rpg_system08

	hide	TreasureKey				;; 鍵を非表示にする
	set		Hero	has_key		1	;; 「キーを持っている」フラグをセット

	text	鍵を手に入れた！
End
